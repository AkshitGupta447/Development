# 1. Initialize the local folder as a Git repository
git init

# 2. Add all files to the staging area
git add .

# 3. Commit your files with a message
git commit -m "Initial commit"

# 4. Link your local repo to GitHub (Replace URL with your copied link)
git remote add origin https://github.com

# 5. Push your code to the 'main' branch
git branch -M main
git push -u origin main
